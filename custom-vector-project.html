Technical Implementation

Core Structure

pub struct RawVec {
    ptr: *mut T,      // Raw pointer to heap memory
    cap: usize,       // Total allocated capacity
}

impl RawVec {
    pub fn new() -> Self {
        RawVec {
            ptr: ptr::null_mut(),
            cap: 0,
        }
    }

    pub fn with_capacity(capacity: usize) -> Self {
        // Handle zero-sized types
        if std::mem::size_of::() == 0 {
            return RawVec {
                ptr: std::ptr::NonNull::dangling().as_ptr(),
                cap: usize::MAX,
            };
        }
        // ... allocation logic ...
    }
}

Iterator Implementation

pub struct IntoIter {
    buf: RawVec,
    start: usize,
    end: usize,
}

pub struct Iter<'a, T> {
    buf: &'a RawVec,
    start: usize,
    end: usize,
}

pub struct IterMut<'a, T> {
    buf: &'a mut RawVec,
    start: usize,
    end: usize,
}

pub struct Drain<'a, T> {
    buf: &'a mut RawVec,
    start: usize,
    end: usize,
    len: usize,
    original_start: usize,
}
